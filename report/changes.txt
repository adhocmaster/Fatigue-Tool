Fatigue is one of the 13 mood states \cite{stone:fatigue}. Stress causes
emotions and mood swings.

  \begin{enumerate}
   		\item Exhausted \& Drowsy
   		\begin{enumerate}
   		  \item Sleepiness
   		  \item Restlessness
   		  \item Feeling tired
   		  \item Sluggish
   		  \item Lethargy
   		\end{enumerate}
   		\item Mental decline
   		\begin{enumerate}
   		  \item Nervousness
   		  \item Unwillingness
   		  \item Lack of focus
   		  \item Absentmindedness
   		  \item Anxiety
   		  \item Weariness
   		\end{enumerate}
   		\item Incongruity in body \& nervous systems
   		\begin{enumerate}
   		  \item Physical Strength
   		  \item Pain in limbs
   		  \item Strain in eyes
   		  \item Dizziness
   		\end{enumerate}
\end{enumerate}
